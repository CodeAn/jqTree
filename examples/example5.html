Example 5 - Load on demand

This examples loads data on demand in the tree.before_open event.

html

<div id="tree1"></div>

javascript

var $tree = $('#tree1');
$tree.tree({
    data: ExampleData.getFirstLevelData()
});
$tree.bind('tree.before_open', function(e) {
    var node = e.node;

    if (! node.hasChildren()) {
        var data = ExampleData.getChildrenOfNode(node.id);
        $tree.tree('loadData', data, node);
    }
});
